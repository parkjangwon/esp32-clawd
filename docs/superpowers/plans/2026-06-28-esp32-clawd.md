- [ ] **Step 3: CLI Receiver 구현 — Unix socket + CLI 모드**

`bridge/src/index.ts`:
```typescript
#!/usr/bin/env node
import { createServer } from 'node:net';
import { existsSync, unlinkSync } from 'node:fs';
import { CliReceiver } from './cli-receiver.js';
import { StateCache } from './state-cache.js';

const SOCK_PATH = '/tmp/clawd.sock';
const VALID_STATES = [
  'idle', 'thinking', 'typing', 'building', 'error', 'happy',
  'sleeping', 'subagent', 'notification', 'sweeping', 'carrying',
  'subagent_multi'
];

const cache = new StateCache();

// Clean stale socket
if (existsSync(SOCK_PATH)) unlinkSync(SOCK_PATH);

const server = createServer((socket) => {
  let buf = '';
  socket.on('data', (data: Buffer) => {
    buf += data.toString();
    const lines = buf.split('\n');
    buf = lines.pop() || '';
    for (const line of lines) {
      try {
        const msg = JSON.parse(line.trim());
        if (!msg.state || !VALID_STATES.includes(msg.state)) {
          console.error(`[clawd] invalid state: ${line}`);
          continue;
        }
        if (cache.isDuplicate(msg.state)) continue;
        cache.setState(msg.state);
        console.log(`[state] ${msg.state}`);
      } catch {
        // non-JSON line → try as raw state name
        const raw = line.trim();
        if (VALID_STATES.includes(raw)) {
          if (cache.isDuplicate(raw)) continue;
          cache.setState(raw);
          console.log(`[state] ${raw}`);
        }
      }
    }
  });
});

server.listen(SOCK_PATH, () => {
  console.log(`[clawd] listening on ${SOCK_PATH}`);
});

// Also support CLI args: clawd state thinking
const args = process.argv.slice(2);
if (args[0] === 'state' && args[1]) {
  const state = args[1];
  if (VALID_STATES.includes(state)) {
    const net = await import('node:net');
    const client = net.createConnection(SOCK_PATH, () => {
      client.write(JSON.stringify({ state }) + '\n');
      client.end();
    });
  }
}
```

`bridge/src/state-cache.ts`:
```typescript
export type ClawdState = string;

export class StateCache {
  private last = '';
  private lastTime = 0;
  private debounceMs = 200;

  isDuplicate(state: ClawdState): boolean {
    if (state === this.last && Date.now() - this.lastTime < this.debounceMs) {
      return true;
    }
    return false;
  }

  setState(state: ClawdState): void {
    this.last = state;
    this.lastTime = Date.now();
  }

  getLast(): ClawdState {
    return this.last;
  }
}
```

`bridge/src/ws-server.ts`:
```typescript
import { WebSocketServer, WebSocket } from 'ws';
import type { ClawdState } from './cli-receiver.js';

export class WsServer {
  private wss: WebSocketServer;
  private clients = new Set<WebSocket>();

  constructor(port: number = 9120) {
    this.wss = new WebSocketServer({ port, host: '0.0.0.0' });
    this.wss.on('connection', (ws) => {
      console.log('[ws] client connected');
      this.clients.add(ws);
      ws.on('close', () => {
        console.log('[ws] client disconnected');
        this.clients.delete(ws);
      });
      ws.on('error', (err) => {
        console.error('[ws] error:', err.message);
        this.clients.delete(ws);
      });
    });
    console.log(`[ws] listening on ws://0.0.0.0:${port}`);
  }

  broadcast(state: ClawdState): void {
    const msg = JSON.stringify({ state });
    for (const ws of this.clients) {
      if (ws.readyState === WebSocket.OPEN) {
        ws.send(msg);
      }
    }
  }
}
```

  try {
    // Try mDNS: clawd.local

#include "wifi.h"
#include <esp_wifi.h>
#include <esp_event.h>
#include <esp_log.h>
#include <nvs_flash.h>
#include <string.h>

static const char *TAG = "wifi";
static bool connected = false;

static void event_handler(void *arg, esp_event_base_t base,
                          int32_t id, void *data) {
    if (base == WIFI_EVENT && id == WIFI_EVENT_STA_START) {
        esp_wifi_connect();
    } else if (base == WIFI_EVENT && id == WIFI_EVENT_STA_DISCONNECTED) {
        connected = false;
        ESP_LOGW(TAG, "disconnected, reconnecting...");
        esp_wifi_connect();
    } else if (base == IP_EVENT && id == IP_EVENT_STA_GOT_IP) {
        connected = true;
        auto *ev = (ip_event_got_ip_t *)data;
        ESP_LOGI(TAG, "connected, IP: " IPSTR, IP2STR(&ev->ip_info.ip));
    }
}

esp_err_t wifi_init_sta(const char *ssid, const char *password) {
    esp_err_t ret = nvs_flash_init();
    if (ret == ESP_ERR_NVS_NO_FREE_PAGES || ret == ESP_ERR_NVS_NEW_VERSION_FOUND) {
        nvs_flash_erase();
        nvs_flash_init();
    }

    esp_netif_init();
    esp_event_loop_create_default();
    esp_netif_create_default_wifi_sta();

    wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
    esp_wifi_init(&cfg);
    esp_event_handler_register(WIFI_EVENT, ESP_EVENT_ANY_ID, event_handler, NULL);
    esp_event_handler_register(IP_EVENT, IP_EVENT_STA_GOT_IP, event_handler, NULL);

    wifi_config_t wifi_cfg = {};
    strncpy((char *)wifi_cfg.sta.ssid, ssid, sizeof(wifi_cfg.sta.ssid) - 1);
    strncpy((char *)wifi_cfg.sta.password, password, sizeof(wifi_cfg.sta.password) - 1);
    wifi_cfg.sta.threshold.authmode = WIFI_AUTH_WPA2_PSK;

    esp_wifi_set_mode(WIFI_MODE_STA);
    esp_wifi_set_config(WIFI_IF_STA, &wifi_cfg);
    esp_wifi_start();
    return ESP_OK;
}
```

#include "websocket_client.h"
#include <esp_websocket_client.h>
#include <esp_log.h>
#include <cjson.h>
#include <string.h>

static const char *TAG = "ws";
static esp_websocket_client_handle_t client = nullptr;
static StateCallback state_cb = nullptr;
static bool connected = false;
static char url_buf[256];

static void on_event(void *arg, esp_event_base_t base, int32_t id, void *data) {
    auto *ev = (esp_websocket_event_data_t *)data;
    switch (id) {
    case WEBSOCKET_EVENT_CONNECTED:
        connected = true;
        ESP_LOGI(TAG, "connected to bridge");
        break;
    case WEBSOCKET_EVENT_DISCONNECTED:
        connected = false;
        ESP_LOGW(TAG, "disconnected");
        break;
    case WEBSOCKET_EVENT_DATA:
        if (ev->data_len > 0 && !ev->op_code) {  // text frame
            cJSON *root = cJSON_ParseWithLength(ev->data_ptr, ev->data_len);
            if (root) {
                cJSON *state = cJSON_GetObjectItem(root, "state");
                if (state && state->valuestring && state_cb) {
                    state_cb(state->valuestring);
                    ESP_LOGI(TAG, "state: %s", state->valuestring);
                }
                cJSON_Delete(root);
            }
        }
        break;
    case WEBSOCKET_EVENT_ERROR:
        ESP_LOGE(TAG, "error");
        break;
    }
}

esp_err_t ws_connect(const char *url, StateCallback on_state) {
    state_cb = on_state;
    strncpy(url_buf, url, sizeof(url_buf) - 1);

    esp_websocket_client_config_t cfg = {};
    cfg.uri = url_buf;
    cfg.disable_auto_reconnect = true;
    cfg.reconnect_timeout_ms = 3000;

    if (client) esp_websocket_client_destroy(client);
    client = esp_websocket_client_init(&cfg);
    esp_websocket_register_events(client, WEBSOCKET_EVENT_ANY, on_event, nullptr);
    esp_websocket_client_start(client);
    return ESP_OK;
}

void ws_poll() {
    if (!connected && client) {
        esp_websocket_client_start(client);
    }
}
```

#include "state_machine.h"

struct StateMap {
    const char *name;
    ClawdState state;
};

static const StateMap kMap[] = {
    {"idle", ClawdState::kIdle},
    {"thinking", ClawdState::kThinking},
    {"typing", ClawdState::kTyping},
    {"building", ClawdState::kBuilding},
    {"error", ClawdState::kError},
    {"happy", ClawdState::kHappy},
    {"sleeping", ClawdState::kSleeping},
    {"subagent", ClawdState::kSubagent},
    {"notification", ClawdState::kNotification},
    {"sweeping", ClawdState::kSweeping},
    {"carrying", ClawdState::kCarrying},
    {"subagent_multi", ClawdState::kSubagentMulti},
};

ClawdState clawd_state_from_string(const std::string &s) {
    for (auto &m : kMap) {
        if (s == m.name) return m.state;
    }
    return ClawdState::kIdle;
}

const char *clawd_state_to_string(ClawdState s) {
    for (auto &m : kMap) {
        if (s == m.state) return m.name;
    }
    return "idle";
}

```c
#include "display.h"
#include <esp_lcd_panel_vendor.h>
#include <esp_lcd_panel_io.h>
#include <esp_lcd_panel_ops.h>
#include <esp_log.h>
#include <driver/spi_master.h>
#include <lvgl.h>

static const char *TAG = "display";

// Waveshare 1.54" pin mapping
#define PIN_CS   21
#define PIN_DC   45
#define PIN_RST  40
#define PIN_MOSI 39
#define PIN_CLK  38
#define PIN_BL   46

#define LCD_WIDTH  240
#define LCD_HEIGHT 240

static esp_lcd_panel_io_handle_t io = nullptr;
static esp_lcd_panel_handle_t panel = nullptr;
static lv_display_t *lv_disp = nullptr;

esp_err_t display_init() {
    // SPI bus
    spi_bus_config_t bus_cfg = {};
    bus_cfg.mosi_io_num = PIN_MOSI;
    bus_cfg.miso_io_num = GPIO_NUM_NC;
    bus_cfg.sclk_io_num = PIN_CLK;
    bus_cfg.quadwp_io_num = GPIO_NUM_NC;
    bus_cfg.quadhd_io_num = GPIO_NUM_NC;
    bus_cfg.max_transfer_sz = LCD_WIDTH * LCD_HEIGHT * 2;
    ESP_ERROR_CHECK(spi_bus_initialize(SPI3_HOST, &bus_cfg, SPI_DMA_CH_AUTO));

    // Panel IO
    esp_lcd_panel_io_spi_config_t io_cfg = {};
    io_cfg.cs_gpio_num = PIN_CS;
    io_cfg.dc_gpio_num = PIN_DC;
    io_cfg.spi_mode = 3;
    io_cfg.pclk_hz = 40 * 1000 * 1000;
    io_cfg.trans_queue_depth = 10;
    io_cfg.lcd_cmd_bits = 8;
    io_cfg.lcd_param_bits = 8;
    ESP_ERROR_CHECK(esp_lcd_new_panel_io_spi(SPI3_HOST, &io_cfg, &io));

    // Panel
    esp_lcd_panel_dev_config_t panel_cfg = {};
    panel_cfg.reset_gpio_num = PIN_RST;
    panel_cfg.rgb_ele_order = LCD_RGB_ELEMENT_ORDER_RGB;
    panel_cfg.bits_per_pixel = 16;
    ESP_ERROR_CHECK(esp_lcd_new_panel_st7789(io, &panel_cfg, &panel));

    esp_lcd_panel_reset(panel);
    esp_lcd_panel_init(panel);
    esp_lcd_panel_invert_color(panel, true);

    // Backlight
    gpio_set_direction(PIN_BL, GPIO_MODE_OUTPUT);
    gpio_set_level(PIN_BL, 1);

    // LVGL
    lv_init();
    lv_disp = lv_display_create(LCD_WIDTH, LCD_HEIGHT);
    lv_display_set_flush_cb(lv_disp, display_flush);

    // Draw buffers
    static lv_color_t buf1[LCD_WIDTH * 16];
    static lv_color_t buf2[LCD_WIDTH * 16];
    lv_display_set_buffers(lv_disp, buf1, buf2, sizeof(buf1), LV_DISPLAY_RENDER_MODE_PARTIAL);

    ESP_LOGI(TAG, "display initialized");
    return ESP_OK;
}

void display_flush(lv_display_t *disp, const lv_area_t *area, uint8_t *px_map) {
    uint32_t w = area->x2 - area->x1 + 1;
    uint32_t h = area->y2 - area->y1 + 1;
    esp_lcd_panel_draw_bitmap(panel, area->x1, area->y1, area->x2 + 1, area->y2 + 1, (uint16_t *)px_map);
    lv_display_flush_ready(disp);
}

lv_display_t *display_get() { return lv_disp; }
```
